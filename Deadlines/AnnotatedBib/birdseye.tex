\subsection*{Matching methods}
\begin{itemize}
\item Probabilistic Record Linkage \citep{fellegi69, arp2018}
\item Prior-informed imputation \citep{Goldstein2012} 
\item Bayesian ``Beta" Record Linkage \citep{sadinle2017}
